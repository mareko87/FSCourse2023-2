<?php get_header(); ?>

<div class="search-page">
    <?php 
    echo '<h1>Rezultat pretraživanja</h1>';
    echo "<hr>";
    // Ovom funkcijom hvatamo parametre pretraživanja
    $s = get_search_query();

    // Funkcijom get_posts() hvatamo rezultat pretraživanja
    $posts = get_posts(array(
        's' => $s,
        'post_type' => 'post',
        'posts_per_page' => 10,
    ));

    // Ispisujemo rezultat pretraživanja
    if ($posts) {
        foreach ($posts as $post) {
            echo '<h2><a href="' . get_permalink($post) . '">' . $post->post_title . '</a></h2>';
            echo '<p>' . $post->post_excerpt . '</p>';
        }
    } else {
        echo '<p>Nema rezultata pretraživanja za "' . $s . '".</p>';
    }
    ?>
</div>
